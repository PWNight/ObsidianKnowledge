pwd - вывести рабочую папку
cd - перейти в папку
cd .. - перейти в предыдущую папку
cd ~ - перейти в корневую папку системы
history - просмотр истории команд в терминале
ls - просмотреть содержимое в папке
ls -a - просмотреть скрытое содержимое в папке
ls -l - содержимое в виде таблицы
ls -la

mkdir - создать папку
tree - вывод дерева папки
touch - создание файла
echo - создание файла
\> файл - создание файла
cat - создание файла
